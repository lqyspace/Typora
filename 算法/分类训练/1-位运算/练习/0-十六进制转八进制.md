        // 第二种方法：取模
        while(n>0){
            System.out.println(dec2Octal(hex2Dec(in.nextLine(), 16), 8));
            n--;
        }

    }

    public static int hex2Dec(String hex, int k){
        int res = 0;
        int flag = 0;
        for(int i=hex.length()-1; i>=0; i--){
            int value = hex.charAt(i);
            int unit = (int)Math.pow(k, flag++);
            if(value>='A'){
                res += unit * (value - 'A' + 10);
            }else {
                res += unit * (value - '0');
            }
        }
        return res;
    }

    public static String dec2Octal(int num, int k){
        Stack<String> stk = new Stack<>();
        StringBuilder sb = new StringBuilder();
        while(num!=0){
            stk.push(String.valueOf(num%k));
            num /= k;
        }
        while (!stk.isEmpty()){
            sb.append(stk.pop());
        }
        return sb.toString();
    }
}
