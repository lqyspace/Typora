class Solution {
    List<List<Integer>> res = new LinkedList<>();

    public List<List<Integer>> permute(int[] nums) {
        List<Integer> trace = new LinkedList<>();
        boolean[] visited = new boolean[nums.length];
        backtrace(nums, trace, visited);
        return res;
    }

    public void backtrace(int[] nums, List<Integer> trace, boolean[] visited){
        if(trace.size()==nums.length){
            res.add(new LinkedList<>(trace));
            return;
        }
        
        for(int i=0; i<nums.length; i++){
            if(visited[i]) continue;
            trace.add(nums[i]);
            visited[i] = true;
            backtrace(nums, trace, visited);
            trace.remove(trace.size()-1);
            visited[i] = false;
        }
    }
}

class Solution {
    List<List<Integer>> res = new LinkedList<>();

    public List<List<Integer>> permute(int[] nums) {
        List<Integer> trace = new LinkedList<>();
        boolean[] visited = new boolean[nums.length];
        for(int i=0; i<nums.length; i++){
            backtrace(nums, trace, visited, i);
        }
        return res;
    }

    public void backtrace(int[] nums, List<Integer> trace, boolean[] visited, int s){
        trace.add(nums[s]);
        visited[s] = true;
        if(trace.size()==nums.length){
            res.add(new LinkedList<>(trace));
            trace.remove(trace.size()-1);
            visited[s] = false;
            return;
        }
        
        for(int i=0; i<nums.length; i++){
            if(visited[i]) continue;
            backtrace(nums, trace, visited, i);
        }
        trace.remove(trace.size()-1);
        visited[s] = false;
    }
}
