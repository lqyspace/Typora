# 227-基本计算器II

```java
class Solution {
    public int calculate(String s) {
		Stack<Integer> stk = new Stack();
        int n = s.length();
        int num=0;
        char sign = '+';
        
        for(int i=0; i<n; i++){
            char c = s.charAt(i);
            if(Character.isDigit(c)){
                num = num*10 + (c - '0');
            }
            
            if(!Character.isDigit(c) && c!=' ' || i==s.length()-1){
                if(sign=='+'){
                    stk.push(num);
                }else if(sign=='-'){
                    stk.push(-num);
                }else if(sign=='*'){
                    stk.push(stk.pop()*num);
                }else if(sign=='/'){
                    stk.push(stk.pop()/num);
                }
                
                num = 0;
                sign = c;
            }
        }
        int ans = 0;
        while(!stk.isEmpty()){
            ans += stk.pop();
        }
        return ans;
    }
}
```
