# 206-反转链表

[力扣206-反转链表](https://leetcode.cn/problems/reverse-linked-list/description/)，难度简单

**迭代：**倒着插入节点

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode reverseList(ListNode head) {
		if(head == null) return head;
        ListNode dummy = new ListNode(-1);
        ListNode a = dummy, b = dummy.next;
        ListNode p = head;
        while(p!=null){
            ListNode tmp = p.next;
            p.next = b;
            a.next = p;
            b = a.next;
            p = tmp;
        }
        return dummy.next;
    }
}
```

```java
    public ListNode reverseList(ListNode head) {
		if(head == null || head.next == null) return head;
        ListNode cur = head, nxt = head;
        ListNode pre = null;

            nxt = cur.next;
           	cur.next = pre;
            pre = cur;
            cur = nxt;
        }
        return pre;
    }
}
```

**递归：**子问题解决原问题

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode reverseList(ListNode head) {
		if(head==null) return head;
        return dfs(head);
    }
    
    public ListNode dfs(ListNode head){
        if(head==null) return head;
        ListNode back = dfs(head.next);
        
        if(back!=null){
            ListNode p = back;
            while(p.next!=null)
            	p = p.next;
            p.next = head;
            head.next = null;
        }
        
        return back==null? head: back;
    }
}
```

**扩展：翻转前N个节点**

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    ListNode successor = null; // 后继节点
    public ListNode reverseList(ListNode head, int n) {
		if(head==null || head.next==null) return head;
        return reverseN(head, n);
    }
    // 反转以 head 为起点的 n 个节点，返回新的头结点
    ListNode reverseN(ListNode head, int n) {
        if (n == 1) {
            // 记录第 n + 1 个节点
            successor = head.next;
            return head;
        }
        // 以 head.next 为起点，需要反转前 n - 1 个节点
        ListNode last = reverseN(head.next, n - 1);

        head.next.next = head;
        // 让反转之后的 head 节点和后面的节点连起来
        head.next = successor;
        return last;
    }

}
```
