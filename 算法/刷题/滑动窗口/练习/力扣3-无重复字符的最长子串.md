# 无重复字符的最长子串

[3-无重复字符的最长子串](https://leetcode.cn/problems/longest-substring-without-repeating-characters/)。难度中等

滑动窗口：

```java
class Solution {
    public int lengthOfLongestSubstring(String s) {
		HashMap<Character, Integer> window = new HashMap<>();
        int left = 0, right = 0;
        int res = 0;
        while(right < s.length()){
            char c = s.charAt(right);
            window.put(c, window.getOrDefault(c, 0)+1);
            right++;
            
            if(window.get(c)>1){
                char l = s.charAt(left);
                window.put(l, window.get(l)-1);
                left++;
            }
            res = Math.max(res, right - left);            
        }
        return res;
    }
}
```

动态规划：

```java
class Solution {
    public int lengthOfLongestSubstring(String s) {
		int n = s.length();
        if(n==0) return 0;
        if(n==1) return 1;// 一定要注意这两个基础条件
        int[] dp = new int[n];// 以i结尾的最长无重复子串
        Arrays.fill(dp, 1);
        int left = 0;
        int res = 0;
        for(int right=1; right<n; right++){
            String sub = s.substring(left, right);
            char c = s.charAt(right);
            if(sub.indexOf(c)==-1)
                dp[right] = dp[right-1] + 1;
            else {
                dp[right] = right - left - sub.indexOf(c);
                left = left + sub.indexOf(c) + 1;
            }
            res = Math.max(res, dp[right]);
        }
        return res;
    }
}
```

```java
class Solution {
    public int lengthOfLongestSubstring(String s) {
        if(s==null) return 0;
        if(s.length()<=1)
            return s.length();
        int[] dp = new int[s.length()];
        int res = -1;
        dp[0] = 1;
        for(int i=1; i<s.length(); i++){
            // 获取子串
            String sub = s.substring(i - dp[i-1], i);
            char c = s.charAt(i);
            int tmp = sub.lastIndexOf(c);
            if(tmp==-1)
                dp[i] = dp[i-1] + 1;
            else
                dp[i] = dp[i-1] - tmp;
            res = Math.max(res, dp[i]);
        }
        return res==-1? 0: res;
    }
}
```

```java
class Solution {
    public int lengthOfLongestSubstring(String s) {
        if(s == null) return 0;
        if(s.length()<=1) 
            return s.length();
        int dp = -1;
        int left = 0, right=1;
        while(right<s.length()){
            String sub = s.substring(left, right);
            char c = s.charAt(right);
            right++;
            if(sub.contains(String.valueOf(c))){
                int ind = sub.indexOf(c);
                left = left + ind + 1;
            }
            dp = Math.max(dp, right - left);
        }
        return dp==-1? 0: dp;
    }
}
```
