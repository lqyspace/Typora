# 54-螺旋矩阵

[力扣54-螺旋矩阵](https://leetcode.cn/problems/spiral-matrix/)，难度中等

![image-20230820065347459](https://raw.githubusercontent.com/lqyspace/mypic/master/PicBed/202308200705562.png)

自己写的代码：

```java
class Solution {
    private int LEFT = 1;
    private int RIGHT = 2;
    private int UP = 3;
    private int DOWN = 4;
    public List<Integer> spiralOrder(int[][] matrix) {
		List<Integer> res = new LinkedList<>();
        int m = matrix.length, n = matrix[0].length;
        int i=0, j=0;
        int direct = 1;
        boolean[][] visited = new boolean[m][n];
        while(res.size()<m*n){
            while(direct == LEFT){
                // 如果到了尽头或者前面已经访问过，则往下走
                if(j>=n || visited[i][j]) 
                    direct = DOWN;
                else {
                    res.add(matrix[i][j]);
                    visited[i][j] = true;
                    j++;
                }
            }
            j--;
            i++;
            while(direct == DOWN){
                // 如果到了尽头或者前面已经访问过，则往右走
                if(i>=m || visited[i][j]) 
                    direct = RIGHT;
                else {
                    res.add(matrix[i][j]);
                    visited[i][j] = true;
                    i++;
                }
            }
            i--;
            j--;
            while(direct == RIGHT){
                // 如果到了尽头或者前面已经访问过，则往右走
                if(j<0 || visited[i][j]) 
                    direct = UP;
                else {
                    res.add(matrix[i][j]);
                    visited[i][j] = true;
                    j--;
                }
            }
            j++;
            i--;
            while(direct == UP){
                // 如果到了尽头或者前面已经访问过，则往右走
                if(i<0 || visited[i][j]) 
                    direct = LEFT;
                else {
                    res.add(matrix[i][j]);
                    visited[i][j] = true;
                    i--;
                }
            }
            i++;
            j++;
        }
        return res;
    }
}
```

官方给的：

```java
class Solution {
    public List<Integer> spiralOrder(int[][] matrix) {
        List<Integer> res = new LinkedList<>();
        int m = matrix.length, n = matrix[0].length;
        int left_bound = 0, right_bound = n-1;
        int upper_bound = 0, lower_bound = m-1;

        while(res.size() < m*n){
            // 向左
            if(upper_bound <= lower_bound){
                for(int j=left_bound; j<=right_bound; j++){
                    res.add(matrix[upper_bound][j]);
                }
                // 上边界下移
                upper_bound++;
            }
            
            // 向下
            if(left_bound<=right_bound){
                for(int i=upper_bound; i<= lower_bound; i++){
                    res.add(matrix[i][right_bound]);
                }
                // 右边界左移
                right_bound--;
            }
            
            // 向右
            if(upper_bound<=lower_bound){
                for(int j=right_bound; j>=left_bound; j--){
                    res.add(matrix[lower_bound][j]);
                }
                // 下边界上移
                lower_bound--;
            }
            
            // 向上
            if(left_bound<=right_bound){
                for(int i=lower_bound; i>=upper_bound; i--){
                    res.add(matrix[i][left_bound]);
                }
                // 左边界右移
                left_bound++;
            }
        }
        return res;
    }
}
```
