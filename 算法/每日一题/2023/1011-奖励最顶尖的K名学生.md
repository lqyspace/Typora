我自己写的代码：哈希集合+哈希表+优先级队列

```java
class Solution {
    public List<Integer> topStudents(String[] positive_feedback, String[] negative_feedback, String[] report, int[] student_id, int k) {
        HashSet<String> pos = new HashSet<>();
        HashSet<String> neg = new HashSet<>();
        List<Integer> res = new ArrayList<>();

        for(String pf: positive_feedback){
            pos.add(pf);
        }
        for(String nf: negative_feedback){
            neg.add(nf);
        }
        
        HashMap<Integer, Integer> hm = new HashMap<>();
        for(int i=0; i<report.length; i++){
            hm.put(student_id[i], 0);
        }

        for(int i=0; i<report.length; i++){
            String phrase = report[i];
            int left = 0;
            while(left < phrase.length()){
                while(left < phrase.length() && !Character.isLetter(phrase.charAt(left))){
                    left++;
                }
                StringBuilder sb = new StringBuilder();
                while(left < phrase.length() && Character.isLetter(phrase.charAt(left))){
                    sb.append(phrase.charAt(left));
                    left++;
                }
                if(sb==null || sb.length()==0){
                    continue;
                }
                if(pos.contains(sb.toString())){
                    hm.put(student_id[i], hm.getOrDefault(student_id[i], 0)+3);
                }else if(neg.contains(sb.toString())){
                    hm.put(student_id[i], hm.getOrDefault(student_id[i], 0)-1);
                }
            }
        }

        PriorityQueue<int[]> pq = new PriorityQueue<>(new Comparator<int[]>(){
            public int compare(int[] a, int[] b){
                return b[1]-a[1]==0? a[0]-b[0]: b[1]-a[1];
            }
        });
        
        for(Map.Entry<Integer, Integer> entry: hm.entrySet()){
            pq.offer(new int[]{entry.getKey(), entry.getValue()});
        }
        for(int i=0; i<k; i++){
            res.add(pq.poll()[0]);
        }
        return res;
    }
}
```

题解代码：哈希表+排序

```java
class Solution {
    public List<Integer> topStudents(String[] positive_feedback, String[] negative_feedback, String[] report, int[] student_id, int k) {
        HashMap<String, Integer> words = new HashMap<>();
        List<Integer> res = new ArrayList<>();

        for(String pf: positive_feedback){
            words.put(pf, 3);
        }
        for(String nf: negative_feedback){
            words.put(nf, -1);
        }
        int n = report.length;
        int[][] A = new int[n][2];

        for(int i=0; i<report.length; i++){
            int score = 0;
            for(String word: report[i].split(" ")){
                score += words.getOrDefault(word, 0);
            }
            A[i] = new int[]{-score, student_id[i]};
        }

        Arrays.sort(A, (a, b)->a[0]-b[0]==0? a[1]-b[1]: a[0]-b[0]);
        for(int i=0; i<k; i++){
            res.add(A[i][1]);
        }
        return res;
    }
}
```
